# =============================================================================
# requirements.txt — Python Dependencies for thru.ai Drive-Through AI
# =============================================================================
#
# Install everything:  pip install -r requirements.txt
#
# NOTE ON GPU SUPPORT:
#   faster-whisper requires NVIDIA GPU + CUDA for GPU acceleration.
#   If you don't have a GPU, change device="cuda" to device="cpu" in main.py.
#   CPU mode is slower but works on any machine.
# =============================================================================

# --- AI / ML Models ---
anthropic            # Anthropic SDK — talk to Claude AI for conversation & order extraction
elevenlabs           # ElevenLabs SDK — text-to-speech with natural human voices
faster-whisper       # GPU-optimised Whisper — local speech-to-text (no API needed)

# --- Audio Capture & Processing ---
sounddevice          # Cross-platform audio recording from microphone (uses PortAudio)
scipy                # Scientific computing — we use scipy.io.wavfile to save WAV files
numpy                # Numerical arrays — audio data is stored as numpy int16 arrays
pyaudio              # Alternative audio backend (sounddevice dependency on some systems)

# --- Web Server (Kitchen Display Backend) ---
flask                # Lightweight web framework — serves the kitchen display API
flask-cors           # CORS middleware — allows cross-origin requests from kitchen display UI

# --- HTTP Client ---
# 'requests' is used by main.py to POST order updates to the Flask server.
# It's a transitive dependency of several packages above, but listed for clarity.
requests
